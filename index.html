// pages/api/chat.js export default async function handler(req, res) { try { if (req.method !== "POST") { return res.status(405).json({ error: "Método no permitido" }); } const { message } = req.body; if (!message) { return res.status(400).json({ error: "Mensaje vacío" }); } // Simulación de respuesta IA const reply = `Pau.AI: ${message}`; return res.status(200).json({ reply }); } catch (error) { console.error("ERROR API:", error); return res.status(500).json({ error: "Error interno del servidor" }); } }
